## I. ReAct Agent — Implementation from Scratch

### 1.1 Core Loop

```python
import re
import json
from typing import Dict, Callable, List, Tuple

class ReActAgent:
    """Complete ReAct Agent implementation — 250 lines"""

    def __init__(self, llm_call: Callable, tools: Dict[str, Callable],
                 max_steps: int = 10, verbose: bool = True):
        self.llm = llm_call          # def llm_call(prompt: str) -> str
        self.tools = tools           # {"search": func, "calc": func}
        self.max_steps = max_steps
        self.verbose = verbose
        self.history: List[Tuple[str, str]] = []

    def run(self, query: str) -> str:
        for step in range(self.max_steps):
            if self.verbose:
                print(f"\n{'='*40}\nStep {step+1}/{self.max_steps}")

            prompt = self._build_prompt(query)
            response = self.llm(prompt)

            if self.verbose:
                print(f"LLM Response:\n{response[:200]}...")

            # Parse Thought and Action
            parsed = self._parse(response)
            if parsed is None:
                self.history.append(("error", "Parsing failed, please retry"))
                continue

            thought, action_type, action_name, action_input = parsed
            self.history.append(("thought", thought))

            if self.verbose:
                print(f"Thought: {thought}")

            # Agent decides to finish
            if action_type == "FINISH":
                return action_input

            # Execute tool
            if action_name not in self.tools:
                obs = f"Error: Tool '{action_name}' does not exist. Available: {list(self.tools.keys())}"
            else:
                try:
                    result = self.tools[action_name](action_input)
                    obs = str(result)
                except Exception as e:
                    obs = f"Tool execution error: {e}"

            self.history.append(("action", f"{action_name}({action_input})"))
            self.history.append(("observation", obs))

            if self.verbose:
                print(f"Action: {action_name}({action_input})")
                print(f"Observation: {obs[:200]}")

        return "Maximum steps reached, task incomplete"

    def _parse(self, response: str):
        """Parse LLM output, extract Thought and Action"""
        # Match Thought
        thought_match = re.search(r"Thought:\s*(.+?)(?=\n(?:Action|Observation|$))",
                                  response, re.DOTALL)
        thought = thought_match.group(1).strip() if thought_match else ""

        # Match FINISH
        finish_match = re.search(r"FINISH\((.+?)\)", response, re.DOTALL)
        if finish_match:
            return (thought, "FINISH", None, finish_match.group(1).strip())

        # Match Action: tool_name(tool_input)
        action_match = re.search(r"Action:\s*(\w+)\((.+?)\)", response)
        if action_match:
            return (thought, "ACTION",
                    action_match.group(1), action_match.group(2).strip())

        return None

    def _build_prompt(self, query: str) -> str:
        tools_desc = "\n".join(
            f"  - {name}: {func.__doc__ or 'No description'}"
            for name, func in self.tools.items()
        )
        history_str = "\n".join(
            f"{role.capitalize()}: {content[:300]}"
            for role, content in self.history[-9:]  # Only keep the last 3 rounds (9 entries)
        )
        return f"""You are an AI assistant that can call tools. Complete tasks using the ReAct pattern.

## Available Tools
{tools_desc}

## History
{history_str if history_str else "(None)"}

## Current Task
{query}

## Output Format (strictly follow)
Thought: [Your thinking about the current situation and next step]
Action: tool_name(parameters)  ← Use when calling a tool
or
FINISH(final answer)  ← Use when the task is complete
"""
```

### 1.2 Tool Definitions

```python
import requests
import math

def search_web(query: str) -> str:
    """Search web for information. Parameter query: search keywords"""
    # Can use DuckDuckGo / SerpAPI etc.
    url = f"https://api.duckduckgo.com/?q={query}&format=json"
    try:
        resp = requests.get(url, timeout=10)
        data = resp.json()
        return data.get("AbstractText", "No relevant information found")
    except:
        return f"Search results: Information about '{query}'..."

def calculate(expression: str) -> str:
    """Safely compute a mathematical expression. Parameter expression: e.g., '2+3*4'"""
    allowed = set("0123456789+-*/().^% ")
    if not all(c in allowed for c in expression):
        return "Error: Expression contains disallowed characters"
    # Safe eval — disable builtins
    result = eval(expression, {"__builtins__": {}},
                  {"sqrt": math.sqrt, "pow": pow, "pi": math.pi})
    return f"{result}"

def read_file(path: str) -> str:
    """Read file content. Parameter path: file path"""
    try:
        with open(path, 'r') as f:
            return f.read(2000)  # Limit length
    except FileNotFoundError:
        return f"Error: File '{path}' does not exist"

    llm_call=lambda p: openai_call(p),  # Replace with actual LLM call
```

## II. RAG Agent with Memory

```python
import numpy as np

class RAGAgent(ReActAgent):
    """Adds vector memory on top of ReAct — Agent can 'remember' previous query results"""

    def __init__(self, *args, embedding_model=None, **kwargs):
        super().__init__(*args, **kwargs)
        self.embed = embedding_model   # text → vector
        self.memory: List[Tuple[np.ndarray, str]] = []  # (embedding, content)

    def remember(self, content: str):
        """Store important information into long-term memory"""
        vec = self.embed(content)
        self.memory.append((vec, content))

    def recall(self, query: str, top_k: int = 3) -> str:
        """Retrieve historical memories most relevant to the query"""
        if not self.memory:
            return "(No relevant memories)"
        q_vec = self.embed(query)
        scores = [np.dot(q_vec, m_vec) for m_vec, _ in self.memory]
        top_indices = np.argsort(scores)[-top_k:][::-1]
        return "\n".join(
            f"- {self.memory[i][1][:200]}" for i in top_indices
        )

    def run(self, query: str) -> str:
        # First check memory
        memory_context = self.recall(query)
        enhanced_query = f"Relevant historical memories:\n{memory_context}\n\nCurrent task: {query}"
        result = super().run(enhanced_query)
        # Store result in memory
        self.remember(f"Q: {query} → A: {result}")
        return result
```

## III. LangGraph — Stateful Graph Agent

```python
from typing import TypedDict, Annotated
import operator

# Define graph state
class AgentState(TypedDict):
    messages: Annotated[list, operator.add]  # Message history (auto-append)
    next_step: str                           # Which node to go to next
    final_answer: str                        # Final answer

# Define nodes
def planner(state: AgentState) -> AgentState:
    """Analyze task → decide whether search is needed"""
    last_msg = state["messages"][-1]
    # LLM judges: needs external info? → search; no → answer
    decision = llm_judge(last_msg)  # "search" or "answer"
    return {"next_step": decision}

def search(state: AgentState) -> AgentState:
    """Execute search → add result to messages"""
    result = search_web(state["messages"][-1])
    return {"messages": [f"Search result: {result}"], "next_step": "planner"}

def answer(state: AgentState) -> AgentState:
    """Synthesize all information → generate final answer"""
    answer = llm_generate(state["messages"])
    return {"final_answer": answer, "next_step": "END"}

# Build graph
from langgraph.graph import StateGraph, END

graph = StateGraph(AgentState)
graph.add_node("planner", planner)
graph.add_node("search", search)
graph.add_node("answer", answer)

graph.set_entry_point("planner")
graph.add_conditional_edges("planner",
    lambda s: s["next_step"],  # → "search" or "answer"
    {"search": "search", "answer": "answer"})
graph.add_edge("search", "planner")  # After search, return to planner to re-evaluate
graph.add_edge("answer", END)

app = graph.compile()
result = app.invoke({"messages": ["What is the weather in Beijing today?"]})
print(result["final_answer"])
```

## IV. Multi-Agent Code Collaboration — AutoGen Style

```python
class CodeAgent:
    """An Agent that can write code, run code, see results, and fix errors"""

    def __init__(self, llm):
        self.llm = llm
        self.workspace = {}

    def solve(self, task: str, max_iterations: int = 5) -> str:
        prompt = f"""You are a Python expert. User task: {task}

Each iteration:
1. Write Python code and execute it
2. Look at the execution result
3. Decide: fix the code / complete the task"""

        for i in range(max_iterations):
            # LLM generates code
            code = self.llm(f"{prompt}\n\nPrevious attempt:\n{self.workspace.get('last_result', 'None')}")
            code = self._extract_code(code)

            # Execute in sandbox
            result = self._execute(code)

            if result["success"] and self._check_complete(task, result):
                return f"Task complete!\nCode:\n{code}\nResult: {result['output']}"

            # Store result, fix in next iteration
            self.workspace["last_result"] = f"Code:\n{code}\nError:\n{result['error']}"

        return "Maximum iterations reached"

    def _execute(self, code: str) -> dict:
        """Execute Python code in a restricted environment"""
        import subprocess, tempfile
        with tempfile.NamedTemporaryFile(mode='w', suffix='.py', delete=False) as f:
            f.write(code)
        try:
            output = subprocess.run(
                ["python3", f.name], capture_output=True, text=True, timeout=30
            )
            return {"success": output.returncode == 0,
                    "output": output.stdout,
                    "error": output.stderr}
        except subprocess.TimeoutExpired:
            return {"success": False, "output": "", "error": "Timeout — possible infinite loop"}
```

## V. Framework Selection Guide

| Scenario | Recommended Framework | Reason |
|----------|---------------------|--------|
| Learning Agent principles | Hand-write ReAct | Control every line, understand the core loop |
| Rapid prototype validation | LangChain | Complete components, rich documentation |
| Complex multi-step logic | LangGraph | Stateful control, debuggable |
| Multi-Agent collaboration | AutoGen / CrewAI | Clear roles, ready-made communication protocols |
| Knowledge-intensive | LlamaIndex | Indexing + retrieval is the core |
| Production deployment | OpenAI Agents SDK | Guardrails + Monitoring + Tracing |
| Productization | Self-developed lightweight framework | Minimal dependencies, optimal performance |

> Final advice: Learning LangGraph is enough to handle 90% of Agent needs. LangGraph is the "ultimate form" of Agent frameworks — its graph structure suits both simple chains and complex multi-Agent collaboration.
