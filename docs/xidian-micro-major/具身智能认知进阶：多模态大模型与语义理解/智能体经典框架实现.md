## 一、ReAct Agent——从零实现

### 核心循环

```python
class ReActAgent:
    def __init__(self, llm, tools):
        self.llm = llm
        self.tools = tools  # {"search": search_func, "calc": calc_func}
        self.history = []

    def run(self, query, max_steps=10):
        prompt = self._build_prompt(query)
        for step in range(max_steps):
            response = self.llm(prompt)
            thought, action_name, action_input = self._parse(response)

            if action_name == "FINISH":
                return action_input

            result = self.tools[action_name](action_input)
            self.history.append(("thought", thought))
            self.history.append(("action", f"{action_name}({action_input})"))
            self.history.append(("observation", result))
            prompt = self._build_prompt(query)
        return "Agent 在步数限制内未完成任务"

    def _build_prompt(self, query):
        tools_desc = "\n".join(
            f"- {name}: {func.__doc__}"
            for name, func in self.tools.items()
        )
        history_str = "\n".join(
            f"{role}: {content}" for role, content in self.history
        )
        return f"""你是可以调用工具的 AI 助手，使用 ReAct 模式。

可用工具：
{tools_desc}

对话历史：
{history_str}

用户问题：{query}

请按以下格式输出：
Thought: [思考]
Action: [工具名]([参数]) 或 FINISH([最终答案])
"""
```

### 工具定义

```python
def search_wikipedia(query: str) -> str:
    """搜索 Wikipedia，参数为搜索词，返回摘要"""
    import requests
    url = f"https://en.wikipedia.org/api/rest_v1/page/summary/{query}"
    return requests.get(url).json().get("extract", "未找到结果")

def calculate(expression: str) -> str:
    """计算数学表达式，返回计算结果"""
    return str(eval(expression, {"__builtins__": {}}, {}))
```

## 二、Function Calling Agent

```python
import json

class FunctionCallingAgent:
    def __init__(self, model, functions):
        self.model = model
        self.functions = functions
        self.messages = []

    def run(self, query):
        self.messages.append({"role": "user", "content": query})
        while True:
            response = self.model.chat(
                messages=self.messages,
                tools=self.functions,
                tool_choice="auto"
            )
            msg = response.choices[0].message
            if msg.tool_calls is None:
                return msg.content  # 模型直接回答

            for tool_call in msg.tool_calls:
                func_name = tool_call.function.name
                func_args = json.loads(tool_call.function.arguments)
                result = self._execute(func_name, func_args)
                self.messages.append({
                    "role": "tool",
                    "tool_call_id": tool_call.id,
                    "content": str(result)
                })

    def _execute(self, name, args):
        registry = {
            "get_weather": lambda a: f"{a['city']}天气: 晴, 28°C",
            "send_email": lambda a: f"已发送至 {a['to']}",
        }
        return registry.get(name, lambda _: "未知工具")(args)
```

## 三、LangChain Agent——一行代码

```python
from langchain.agents import initialize_agent, AgentType
from langchain.tools import Tool

tools = [
    Tool(name="Search", func=search_wikipedia,
         description="搜索 Wikipedia"),

agent = initialize_agent(
```

## 四、Multi-Agent——Swarm 风格

```python
from dataclasses import dataclass
from typing import List

@dataclass
class Agent:
    name: str
    instruction: str
    functions: List[callable]

class Swarm:
    def __init__(self, agents: List[Agent]):
        self.agents = {a.name: a for a in agents}
        self.messages = []

    def run(self, query: str, start_agent: str = "triage"):
        current = self.agents[start_agent]
        self.messages.append({"role": "user", "content": query})
        while True:
            response = self._call_llm(current)
            if response.get("action") == "transfer":
                current = self.agents[response["target"]]
                self.messages.append({
                    "role": "system",
                    "content": f"已切换到 {current.name}"
                })
                continue
            return response["content"]

# 客服 Swarm
triage = Agent("triage", "分类问题并转给合适的 Agent", [])
tech = Agent("tech", "处理技术问题", [])
billing = Agent("billing", "处理退款/订阅", [])
swarm = Swarm([triage, tech, billing])
```

## 五、各框架对比

| 维度 | ReAct 手写 | LangChain | CrewAI/Swarm | OpenAI Assistants |
|------|-----------|-----------|-------------|-----------------|
| 控制力 | 最高 | 中 | 中 | 低 |
| 上手速度 | 慢 | 快 | 中 | 最快 |
| 灵活性 | 极高 | 高 | 中 | 低 |
| 生产就绪 | 需自己加 | 部分 | 否 | ✅ |
| 学习价值 | ⭐⭐⭐⭐⭐ | ⭐⭐⭐ | ⭐⭐⭐ | ⭐⭐ |

> 学习建议：先用 ReAct 手写一遍（理解核心循环）→ 用 LangChain（理解工程抽象）→ 最后用 Swarm（理解多 Agent 协作）。
